Examples

>>> from pyedb import Edb
>>> edb = Edb()
>>> freq = [0, 2, 3, 4, 5, 6]
>>> rel_perm = [1e9, 1.1e9, 1.2e9, 1.3e9, 1.5e9, 1.6e9]
>>> loss_tan = [0.025, 0.026, 0.027, 0.028, 0.029, 0.030]
>>> diel = edb.materials.add_multipole_debye_material("My_MP_Debye", freq, rel_perm, loss_tan)
